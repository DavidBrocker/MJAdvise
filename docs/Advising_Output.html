# Preprocessing Function 
prepocess <- function(file,csv = TRUE){
  library(dplyr)
  # Assuming File is a CSV
  file <- read.csv(file.choose())
  # Otherwise, read as an XLSX
  if(csv == FALSE) {
    file <- readxl::read_excel(file.choose())
  }
  print(str(file))
  
  # Clean Names
  file_cln <- 
    file |> 
    janitor::clean_names()
  
  # Skim Numeric Data
  file_cln |> 
    select(where(is.numeric)) |> 
    skimr::skim() |>
    skimr::yank("numeric")
  
}

# Correlation Hux Wrapper
hux_corr <- function(ct){
  if(!class(ct) == "htest"){
    print("Source must be of class 'htest'")
  } else{
    ct |> 
      broom::tidy() |> 
      mutate(
        `CI[LL,UU]` = paste0("[",conf.low |> round(2),
                             ",",conf.high |> round(2),
                             "]")
      ) |> 
      rename(
        r = estimate,
        t = statistic, 
        p = p.value, 
        df = parameter,
      ) |> 
      relocate(
        p, .after = df
      ) |>
      relocate(
        alternative, .after = last_col()
      ) |> 
      select(-conf.low,-conf.high,-method) |> 
      hux() |> 
      theme_article()
  }
}

Exploratory Analyses: Methodology

Data was collapsed such that each student has a total summary count of:

  • Emails sent by their advisor

  • Emails replied to by the student - phone calls made to the student by their advisor

  • Duration of the phone calls

  • A variable, gpa_diff was calculated that subtracts the studentâ€™s semester end GPA from the GPA they started with. If the values are greater than 0, the student improved. If the value is less than 0, the student did not improve.

In all instance (save for freshman data) the time-frame was between January and May.
